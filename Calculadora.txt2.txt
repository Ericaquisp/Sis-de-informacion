//Funcion para quitar el .0
public String cero(Double resultado){
    String retorno = "";
    
    retorno = Double.toString(resultado);

    if(resultado % 1 == 0){
        retorno = retorno.substring(0, retorno.length() - 2);
    }

    return retorno;
}
